IDENTIFY TYPE :
char :
	-string size 1
	-not a number

int :
	-only numbers & starting '-'
	-check length for more less then int min/max length
	-inferior to int max
	-if '-' = superior to int min
	-no '.' (decimals)

float :
	-"-inff" | "+inff" | "nanf" | only numbers & '.' & starting '-'
	-inferior to float max
	-if '-' = superior to float min
	-there is ONE '.' (decimal)
	-can finish by '.f'

double :
	-"-
	" | "+inf" | "nan" | only numbers & '.' & starting '-'
	-inferior to double max
	-if '-' = superior to double min
	-there is ONE '.' (decimal)
	-can finish by '.'
----------------------------------------------------------------
CONVERT :
use string stream for first convertion and then do casts between types.
	-first convertion with string stream
	-second convertion with casts
-----------------------------------------------------------------
PRINT EVERYTHING :
TYPE CHAR :
	print : char | int | float | double
TYPE INT :
	print : int float double
		if int isprint : print char
	non printable :
		if int is not isprint : char non printable
TYPE FLOAT :
	impossible :
		if float nanf | +inff | -inf : int|char impossible
	non printable :
		if float is not isprint : char non printable
	print : float | double
		if float is print : print char
		if int min < float < int max : print int
TYPE DOUBLE :
	impossible :
		if double nan | +inf | -inf : int|char impossible
		if float min > double > float max : float impossible
		if int min > double > int max : int impossible
	non printable :
		if double is not isprint : char non printable
	print : double
		if float min < double < float max : print float
		if double nan | +inf | -inf : print float
		if int min < double < int max : print int
		if double isprint : print char
TYPE UNDEFINED :
	impossible : char | int | float | double
----------------------------------------------------------------
char :
	PRINT:
		-type int and is printable
		-type float cast to int is printable
		-type double cast to int is printable
		-type char
	IMPOSSIBLE:
		-type float and nanf | +inff | -inff
		-type double and nan | +inf | -inf
	NON PRINTABLE:
		-type int and non printable
		-type float non printable
		-type double non printable
		-type undefined

int :
	PRINT:
		-type char
		-type int
		-type float < int max | > int min
		-type double < int max | > int min
	IMPOSSIBLE:
		-type float and nanf | +inff | -inff
		-type double and nan | +inf | -
		-type undefined

float :
	PRINT:
		-type char
		-type int
		-type float
		-type double < float max | > float min
		-type double nan | +inf | -inf
	IMPOSSIBLE:
		-type double > float max | < float min (not inf or nan)
		-type undefined

double :
	PRINT:
		-type char
		-type int
		-type float
		-type double
	IMPOSSIBLE:
		-type undefined
